Testing 1: Whether the average response time across complaint types is similar or not.¶

In [45]:
# Convert Request_Closing_Time to hours and filter out NaN values
df_test1 = df_clean.dropna(subset=['Request_Closing_Time', 'Complaint Type']).copy()
df_test1['Request_Closing_Time_Hours'] = df_test1['Request_Closing_Time'].dt.total_seconds() / 3600

# Group data by Complaint Type
groups = [df_test1[df_test1['Complaint Type'] == complaint]['Request_Closing_Time_Hours'] 
          for complaint in df_test1['Complaint Type'].unique()]

# Perform one-way ANOVA
f_stat, p_value = f_oneway(*groups)

# Printing the  results
print("Test 1: ANOVA for Average Response Time Across Complaint Types")
print("Null Hypothesis (H0): Average response time is the same across all complaint types.")
print("Alternate Hypothesis (H1): Average response time differs across at least one pair of complaint types.")
print(f"F-Statistic: {f_stat:.2f}")
print(f"P-Value: {p_value:.4f}")

# Interpret results
alpha = 0.05
if p_value < alpha:
    print(f"Since p-value ({p_value:.4f}) < {alpha}, reject the Null Hypothesis.")
    print("Conclusion: The average response time differs significantly across complaint types.")
else:
    print(f"Since p-value ({p_value:.4f}) >= {alpha}, fail to reject the Null Hypothesis.")
    print("Conclusion: No significant difference in average response time across complaint types.")
